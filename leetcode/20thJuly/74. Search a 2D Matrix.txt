class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        row=len(matrix)
        col=len(matrix[0])

        st=0
        last=(row*col)-1

        while st<=last:
            mid=(st+last)//2
            r=mid//col
            c=mid%col

            val=matrix[r][c]

            if val==target:
                return True
            
            elif val<target:
                st=mid+1
            
            else:
                last=mid-1

        return False
