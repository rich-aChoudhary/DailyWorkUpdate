class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        g.sort()
        s.sort()

        g_i=0
        s_i=0
        while s_i<len(s) and g_i<len(g):
            if s[s_i]>=g[g_i]:
                s_i+=1
                g_i+=1
            else:
                s_i+=1

        return g_i
