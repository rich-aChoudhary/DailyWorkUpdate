class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort(key=lambda x:x[1])
        count=1
        last=intervals[0][1]

        for i in range(len(intervals)):
            if intervals[i][0]>=last:
                count+=1
                last=intervals[i][1]

        return len(intervals)-count
