class Solution:
    def majorityElement(self, nums: List[int]) -> List[int]:
        fre={}
        n=len(nums)
        res=[]

        for num in nums:
            fre[num]=fre.get(num,0)+1

        for num,val in fre.items():
            if val>(n//3):
                res.append(num)
        return res
