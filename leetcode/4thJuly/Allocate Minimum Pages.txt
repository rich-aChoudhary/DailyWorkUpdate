class Solution:
    def findPages(self, arr, k):
        # code here
        if k>len(arr):
            return -1
            
        def getSt(pages):
            student=1
            total=0
            for num in arr:
                if (total+num)<=pages:
                    total+=num
                else:
                    student+=1
                    total=num
            return student
        
        low=max(arr)
        high=sum(arr)
        
        while low<=high:
            mid=(low+high)//2
            student=getSt(mid)
            if student>k:
                low=mid+1
            else:
                high=mid-1
        return low
