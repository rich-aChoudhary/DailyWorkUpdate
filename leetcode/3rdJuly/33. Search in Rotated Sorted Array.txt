class Solution:
    def search(self, nums: List[int], target: int) -> int:
        start=0
        last=len(nums)-1

        while start<=last:
            mid=(start+last)//2

            if nums[mid]==target:
                return mid

            #if left half is sorted
            if nums[start]<=nums[mid]:
                if nums[start]<=target<nums[mid]:
                    last=mid-1
                else:
                    start=mid+1

            else:  #if right half is sorted
                if nums[mid]<target<=nums[last]:
                    start=mid+1
                else:
                    last=mid-1

        return -1
