# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        """ Two Pass Approach
        
        # count total nodes
        temp=head
        count=0
        while temp:
            temp=temp.next
            count+=1

        # if 1st node needs removal
        if count==n:
            return head.next

        # Find node before target
        k=count-n
        cur=head
        for i in range(1,k):
            cur=cur.next
            
        # Delete Node.next
        cur.next=cur.next.next

        return head

        """

        # single pass Approach using two pointers
        dummy=ListNode(0,head)

        slow=dummy
        fast=dummy

        for i in range(n):
            fast=fast.next

        while fast.next:
            slow=slow.next
            fast=fast.next

        slow.next=slow.next.next

        return dummy.next
