class Solution:
    def reverse(self, x: int) -> int:
        #for negative sign handling
        sign=-1 if x<0 else 1

        #make every x positive
        x=abs(x)

        res=0

        while x:
            rem=x%10
            res=(res*10)+rem
            x=x//10
        
        #update sign
        res*=sign

        #handle range
        if res<-(2**31) or res>(2**31-1):
            return 0

        return res
