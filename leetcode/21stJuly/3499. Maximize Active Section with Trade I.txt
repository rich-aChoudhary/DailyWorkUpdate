class Solution:
    def maxActiveSectionsAfterTrade(self, s: str) -> int:
        n=len(s)
        active=s.count('1')

        inactive=[]
        i=0
        while i<n:
            if s[i]=='0':
                start=i
                while i<n and s[i]=='0':
                    i+=1
                inactive.append(i-start)
            else:
                i+=1

        maxPairSum=0
        for i in range(1,len(inactive)):
            maxPairSum=max(maxPairSum,inactive[i]+inactive[i-1])
        return maxPairSum+active
