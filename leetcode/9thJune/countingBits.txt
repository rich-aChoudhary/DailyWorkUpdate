class Solution:
    def countBits(self, n: int) -> List[int]:
        """ Brute Force O(nlogn)
        res=[]

        for i in range(n+1):
            ans=self.count1(i)
            res.append(ans)

        return res

    def count1(self,i):
        count=0
        while i:

        #Optimal O(n) using DP
        dp=[0]*(n+1)

        for i in range(1,n+1):
