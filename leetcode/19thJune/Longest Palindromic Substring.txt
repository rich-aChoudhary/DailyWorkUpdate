class Solution:
    def longestPalindrome(self, s: str) -> str:
        ans=""

        def check(st,i,j):
            nonlocal ans
            while i>=0 and j<len(s) and s[i]==s[j]:
                i-=1
                j+=1

            pal=st[i+1:j]
            if len(pal)>len(ans):
                ans=pal

        for i in range(len(s)):
            check(s,i,i)
            check(s,i,i+1)

        return ans
