Rearrange Characters to Make Target String

class Solution:
    def rearrangeCharacters(self, s: str, target: str) -> int:
        fqS={}

        for ch in s:
            fqS[ch]=fqS.get(ch,0)+1

        fqTar={}

        for ch in target:
            fqTar[ch]=fqTar.get(ch,0)+1

        res=len(s)

        for c in target:
            res=min(res,fqS.get(c,0)//fqTar[c])

        return res
