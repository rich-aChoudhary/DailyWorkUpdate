class Solution:
    def missingNumber(self, nums: List[int]) -> int:
        """ Gauss Approach
        n=len(nums)
        sum=0
        total=n*(n+1)//2

        for i in nums:
            sum+=i

        return total-sum
        """

        #XOR approach
        res=len(nums)

        for i,num in enumerate(nums):
            res^=i^num
        return res
