class Solution:
    def pathExistenceQueries(self, n: int, nums: List[int], maxDiff: int, queries: List[List[int]]) -> List[bool]:
        res=[]
        pre=[0]*n
        for i in range(1,n):
            pre[i]=pre[i-1]
            if nums[i]-nums[i-1]>maxDiff:
                pre[i]+=1
        
        for st,en in queries:
            if pre[st]==pre[en]:
                res.append(True)
            else:
                res.append(False)

        return res
