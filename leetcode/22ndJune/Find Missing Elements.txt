class Solution:
    def findMissingElements(self, nums: List[int]) -> List[int]:
        """  using List take O(n^2)
        nums.sort()

        low=nums[0]
        high=nums[len(nums)-1]
        
        res=[]

        for i in range(low,high+1):
            if i not in nums:
                res.append(i)

        return res   """
        #using set O(n)

        low=min(nums)
        high=max(nums)

        numSet=set(nums)

        res=[]

        for i in range(low,high+1):
            if i not in numSet:
                res.append(i)

        return res
