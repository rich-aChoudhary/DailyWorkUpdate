class LRUCache:

    def __init__(self, capacity: int):
        self.map={}
        self.st=[]
        self.n=capacity
        
    def get(self, key: int) -> int:
        if key not in self.map:
            return -1
        self.st.remove(key)
        self.st.append(key)
        return self.map[key]
        
    def put(self, key: int, value: int) -> None:
        if key in self.map:
            self.st.remove(key)
        elif len(self.map)==self.n:
            val=self.st.pop(0)
            self.map.pop(val)
        self.map[key]=value  
        self.st.append(key)
