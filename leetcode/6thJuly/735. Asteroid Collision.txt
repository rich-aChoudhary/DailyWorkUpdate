class Solution:
    def asteroidCollision(self, asteroids: List[int]) -> List[int]:
        st=[]
        for ast in asteroids:
            if ast>0:
                st.append(ast)
                continue
            
            while st and st[-1]>0:
                if st[-1]<abs(ast):
                    st.pop()
                    continue
                elif st[-1]==abs(ast):
                    st.pop()
                break
            else:
                st.append(ast)
        return st
