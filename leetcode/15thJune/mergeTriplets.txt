class Solution:
    def mergeTriplets(self, triplets: List[List[int]], target: List[int]) -> bool:
        ans=set()
        provider={}

        for t in triplets:
            if t[0]>target[0] or t[1]>target[1] or t[2]>target[2]:
                continue

            for i,v in enumerate(t):
                if v==target[i]:
                    ans.add(i)
                    provider[i]=t
        print(provider)
        return len(ans)==3

        #here provider is used for storing exactly which triplet is contributing to raech the target
