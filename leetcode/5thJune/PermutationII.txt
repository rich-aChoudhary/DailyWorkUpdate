class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        res=[]

        def per(nums,path):
            if not nums:
                res.append(path)
                return
            
            unique=set()

            for i in range(len(nums)):
                if nums[i] in unique:
                    continue
                unique.add(nums[i])
                newIp=nums[:i]+nums[i+1:]
                newOp=path+[nums[i]]
                per(newIp,newOp)

        per(nums,[])
        return res
