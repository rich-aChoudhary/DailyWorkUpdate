class Solution:
    def rob(self, nums: List[int]) -> int:
        if len(nums)==1:
            return nums[0]

        def rob1(arr):
            n=len(arr)
            dp=[0]*(n+1)
            dp[0]=0
            dp[1]=arr[0]

            for i in range(2,n+1):
                dp[i]=max(arr[i-1]+dp[i-2],dp[i-1])

            return dp[n]

        return max(rob1(nums[:-1]),rob1(nums[1:]))
