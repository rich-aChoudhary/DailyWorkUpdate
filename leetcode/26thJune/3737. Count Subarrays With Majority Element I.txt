class Solution:
    def countMajoritySubarrays(self, nums: List[int], target: int) -> int:
        i,count=0,0
        while i<len(nums):
            freq=0
            for j in range(i,len(nums)):
                if nums[j]==target:
                    freq+=1

                if (j-i+1)//2<freq:
                    count+=1

            i+=1
        return count
